@st.cache_data(show_spinner=False)
def read_csv(path: Path) -> pd.DataFrame:
    if path.exists():
        try:
            return pd.read_csv(path)
        except UnicodeDecodeError:
            return pd.read_csv(path, encoding="utf-8-sig")
        except Exception:
            return pd.DataFrame()
    return pd.DataFrame()


@st.cache_resource(show_spinner=False)
def read_pickle(path: Path):
    if path.exists():
        try:
            return joblib.load(path)
        except Exception:
            return None
    return None


@st.cache_resource(show_spinner=False)

        if obj is not None:

    pipe = read_pickle(PIPELINE / "pipeline.pkl")

def fmt_num(x, decimals: int = 0) -> str:
    try:
        v = float(x)
        if np.isnan(v):
            return "—"
        return f"{v:,.{decimals}f}"
    except Exception:
        return "—"


def fmt_int(x) -> str:
    return fmt_num(x, 0)


def fmt_money(x) -> str:
    try:
        v = float(x)
        if np.isnan(v):
            return "—"
        return f"${v:,.0f}"
    except Exception:
        return "—"


def fmt_pct(x) -> str:
    try:
        v = float(x)
        if np.isnan(v):
            return "—"
        return f"{v*100:.2f}%"
    except Exception:
        return "—"

def model_rank_table(model_cmp_df: pd.DataFrame) -> pd.DataFrame:
    if model_cmp_df.empty:
        return pd.DataFrame()
    df_rank = model_cmp_df.copy()
    metric_cols = [c for c in ["accuracy", "precision", "recall", "f1_score", "roc_auc"] if c in df_rank.columns]
    for m in metric_cols:
        df_rank[f"{m}_rank"] = df_rank[m].rank(ascending=False, method="dense").astype(int)
    if metric_cols:
        df_rank["average_rank"] = df_rank[[f"{m}_rank" for m in metric_cols]].mean(axis=1)
        sort_cols = ["average_rank"] + (["roc_auc"] if "roc_auc" in df_rank.columns else [])
        asc = [True] + ([False] if "roc_auc" in df_rank.columns else [])
        df_rank = df_rank.sort_values(sort_cols, ascending=asc)
    return df_rank


def plot_tuning_heatmap(tuning_df: pd.DataFrame):
    if tuning_df.empty:
        return empty_fig("Missing tuning_result.csv")

    score_col = safe_col(tuning_df, ["mean_test_score", "rank_test_score"])
    if score_col is None:
        return empty_fig("Tuning file does not contain mean_test_score")

    possible_x = safe_col(tuning_df, ["param_max_depth", "param_n_estimators", "param_learning_rate"])
    possible_y = safe_col(tuning_df, ["param_learning_rate", "param_n_estimators", "param_subsample"])

    if possible_x is None or possible_y is None or possible_x == possible_y:
        view = tuning_df.sort_values(score_col, ascending=(score_col == "rank_test_score")).head(15)
        fig = px.bar(view, x=score_col, y=view.index.astype(str), orientation="h", color=score_col, color_continuous_scale="YlGnBu")
        return theme(fig, "Top Hyperparameter Candidates", 520)

    heat = tuning_df.copy()
    heat[possible_x] = heat[possible_x].astype(str)
    heat[possible_y] = heat[possible_y].astype(str)

    pivot = heat.pivot_table(index=possible_y, columns=possible_x, values=score_col, aggfunc="mean")
    fig = px.imshow(pivot, text_auto=".4f", color_continuous_scale="YlGnBu", aspect="auto")
    fig.update_xaxes(title=possible_x)
    fig.update_yaxes(title=possible_y)
    return theme(fig, "Hyperparameter Tuning Heatmap", 430)


def build_confusion_from_scored(scored_df: pd.DataFrame):
    if scored_df.empty or "loan_status" not in scored_df.columns:
        return None
    pred_col = None
    for c in ["model_prediction", "prediction"]:
        if c in scored_df.columns:
            pred_col = c
            break
    if pred_col is None:
        if "probability_of_default" in scored_df.columns:
            pred = (scored_df["probability_of_default"] >= best_threshold()).astype(int)
        else:
            return None
    else:
        pred = scored_df[pred_col].astype(int)
    y = scored_df["loan_status"].astype(int)
    tn = int(((y == 0) & (pred == 0)).sum())
    fp = int(((y == 0) & (pred == 1)).sum())
    fn = int(((y == 1) & (pred == 0)).sum())
    tp = int(((y == 1) & (pred == 1)).sum())
    return pd.DataFrame([[tn, fp], [fn, tp]], index=["Actual Non-default", "Actual Default"], columns=["Predicted Non-default", "Predicted Default"])

def pca_variance_summary(pca_var_df: pd.DataFrame):
    if pca_var_df.empty:
        return None
    cols = pca_var_df.columns.tolist()
    comp_col = safe_col(pca_var_df, ["principal_component", "component", "PC", cols[0]])
    ratio_candidates = ["explained_variance_ratio", "variance_ratio", "explained_variance"]
    if len(cols) > 1:
        ratio_candidates.append(cols[1])
    ratio_col = safe_col(pca_var_df, ratio_candidates)
    cum_candidates = ["cumulative_variance_ratio", "cumulative_variance", "cumulative"]
    if cols:
        cum_candidates.append(cols[-1])
    cum_col = safe_col(pca_var_df, cum_candidates)
    return comp_col, ratio_col, cum_col

def plot_error_analysis(scored_df: pd.DataFrame):
    if scored_df.empty or "loan_status" not in scored_df.columns:
        return empty_fig("Need loan_status and prediction data")
    temp = scored_df.copy()
    if "model_prediction" not in temp.columns:
        if "probability_of_default" in temp.columns:
            temp["model_prediction"] = (temp["probability_of_default"] >= best_threshold()).astype(int)
        else:
            return empty_fig("Need model_prediction or probability_of_default")
    def error_type(row):
        actual = int(row["loan_status"])
        pred = int(row["model_prediction"])
        if actual == 0 and pred == 0:
            return "True Negative"
        if actual == 0 and pred == 1:
            return "False Positive"
        if actual == 1 and pred == 0:
            return "False Negative"
        return "True Positive"
    temp["error_type"] = temp.apply(error_type, axis=1)
    counts = temp["error_type"].value_counts().rename_axis("error_type").reset_index(name="customers")
    fig = px.bar(
        counts,
        x="error_type",
        y="customers",
        color="error_type",
        text="customers",
        color_discrete_map={"True Negative": "#15803d", "True Positive": "#0e7490", "False Positive": "#d8b45f", "False Negative": "#b91c1c"},
    )
    fig.update_traces(textposition="outside", cliponaxis=False)
    return theme(fig, "Prediction Error Analysis", 400)

    temp["risk_decile"] = pd.qcut(temp["probability_of_default"].rank(method="first"), 10, labels=[f"D{i}" for i in range(10, 0, -1)])
    decile = temp.groupby("risk_decile", observed=False).agg(customers=("loan_status", "count"), default_rate=("loan_status", "mean"), avg_pd=("probability_of_default", "mean")).reset_index()
    fig = go.Figure()
    fig.add_trace(go.Bar(x=decile["risk_decile"], y=decile["default_rate"], name="Actual default rate", text=decile["default_rate"], texttemplate="%{text:.2%}", textposition="outside"))
    fig.add_trace(go.Scatter(x=decile["risk_decile"], y=decile["avg_pd"], mode="lines+markers", name="Average predicted PD", line=dict(width=4)))
    fig.update_yaxes(tickformat=".0%", title="Rate")
    fig.update_xaxes(title="Risk decile, D10 = highest predicted risk")
    return theme(fig, "Risk Decile / Lift Chart", 430)

def safe_columns(dataframe: pd.DataFrame, cols: list[str]) -> list[str]:
    return [c for c in cols if c in dataframe.columns]


def best_threshold() -> float:
    if "best_threshold" in A:
        try:
            return float(A["best_threshold"])
        except Exception:
            return 0.71
    return 0.71


def best_model_name() -> str:
    if "best_model_name" in A:
        return str(A["best_model_name"])
    if not model_cmp.empty and "model" in model_cmp.columns:
        return str(model_cmp.sort_values("roc_auc", ascending=False).iloc[0]["model"])
    return "XGBoost"


def calculate_credit_score(prob_default: float) -> int:
    score = 850 - prob_default * 550
    score = max(300, min(850, score))
    return int(round(score))


def classify_risk_level(score: int) -> str:
    if score >= 700:
        return "LOW"
    elif score >= 600:
        return "MEDIUM"
    return "HIGH"


def make_decision(prob_default: float, score: int) -> str:
    if prob_default < 0.25 and score >= 720:
        return "APPROVE"
    elif prob_default < 0.60 and score >= 500:
        return "REVIEW"
    return "REJECT"


def build_model_input(raw: dict) -> Optional[pd.DataFrame]:
    model_columns = A.get("model_columns")
    if model_columns is None:
        return None

    input_df = pd.DataFrame([raw])

    # One-hot encoding tương thích với data_feature.csv
    input_df = pd.get_dummies(input_df)

    # Reindex đúng columns Person 2 đã lưu
    input_df = input_df.reindex(columns=model_columns, fill_value=0)
    return input_df


def predict(raw: dict) -> Optional[dict]:
    model = A.get("best_model")
    if model is None:
        return None

    X = build_model_input(raw)
    if X is None:
        return None

    try:
        prob_default = float(model.predict_proba(X)[0][1])
    except Exception:
        return None

    score = calculate_credit_score(prob_default)
    risk = classify_risk_level(score)
    decision = make_decision(prob_default, score)

    return {
        "probability_of_default": prob_default,
        "prediction": int(prob_default >= best_threshold()),
        "credit_score": score,
        "risk_level": risk,
        "decision": decision,
    }


def ensure_scoring_columns(dataframe: pd.DataFrame) -> pd.DataFrame:
    data = dataframe.copy()

    if "probability_of_default" not in data.columns:
        data["probability_of_default"] = np.nan

    if "credit_score" not in data.columns:
        data["credit_score"] = data["probability_of_default"].apply(
            lambda x: calculate_credit_score(x) if pd.notna(x) else np.nan
        )

    if "risk_level" not in data.columns:
        data["risk_level"] = data["credit_score"].apply(
            lambda x: classify_risk_level(int(x)) if pd.notna(x) else "UNKNOWN"
        )

    if "decision" not in data.columns:
        data["decision"] = data.apply(
            lambda r: make_decision(
                float(r["probability_of_default"]),
                int(r["credit_score"]),
            )
            if pd.notna(r["probability_of_default"]) and pd.notna(r["credit_score"])
            else "UNKNOWN",
            axis=1,
        )

    return data
